def count_cal
        sum = 0
        x = @list1.head
        y = @list2.head
        while(x != nil && y != nil)
                sum += (x.data.energ * y.data)/100
                x = x.next
                y = y.next
        end
        if(x != nil || y != nil)
                return "error"
        end
        return sum.round(2)
end

def huellanutricional
        i_energia = list1.zip(list2).collect do |alimento, cantidad|
        
                if((alimento.energ * cantidad / 100) < 670)
                        1.0
                elsif((alimento.energ * cantidad / 100 ) < 830)
                        2.0
                else
                        3.0
                end
        end

        i_gei = list1.zip(list2).collect do |alimento, cantidad|

                if((alimento.gei * cantidad / 100) < 800)
                        1.0
                elsif((alimento.gei * cantidad / 100) < 1200)
                        2.0
                else
                        3.0
                end
        end

        i_total_energia = i_energia.sum
        i_total_gei = i_gei.sum

        return ((i_total_energia + i_total_gei) / (list1.count*2)).round(2)



end

def to_s

def impactoambientalh()
                sum1 = self.inject(0){|sum, x| sum + x.energ}
        
                sum2 = self.inject(0){|sum, x| sum + x.prot}
        
                sum3 = self.inject(0){|sum, x| sum + x.gei}
        
                sum4 = self.inject(0){|sum, x| sum + x.terreno}

                cad = String.new
                if ( sum1 < 3000.0)
                        cad = "No se han llegado a las calorías mínimas diarias para un hombre de emtre 20-39 años"
                else 
                        cad = "Se han superado las calorías mínimas diarias para un hombre de entre 20-39 años"
                end
                cad2 = String.new
                if( sum2 < 54.0)
                        cad2 = "no se ha llegado a las proteínas diarias"
                else
                        cad2 = "se ha llegado a las proteínas diarias"
                end  
        
                "#{cad} (#{sum1} de 3000), #{cad2} (#{sum2} de 54) y se tiene un impacto ambiental de #{sum3} kgCO2eq y #{sum4} m2 al año" 
        end
